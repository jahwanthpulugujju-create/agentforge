### **1. Backend: Ollama Client**

```python
# backend/core/llm_router.py
import os
import requests
from typing import Optional, Dict, Any

class LLMRouter:
    """Route requests to local Ollama or cloud APIs based on availability."""
    
    def __init__(self):
        self.ollama_host = os.getenv("OLLAMA_HOST", "http://localhost:11434")
        self.gemini_key = os.getenv("GEMINI_API_KEY")
        self.use_local = os.getenv("USE_LOCAL_MODELS", "false").lower() == "true"
        
        # Model mapping: agent -> (local_model, cloud_model)
        self.model_map = {
            "Architect": ("llama3:8b", "gemini-1.5-pro"),
            "Coder": ("codellama:7b", "gemini-1.5-flash"),
            "Security": ("codellama:7b", "gemini-1.5-flash"),
            "Performance": ("llama3:8b", "gemini-1.5-flash"),
            "Reviewer": ("llama3:8b", "gemini-1.5-flash"),
            "Devil's Advocate": ("llama3:8b", "gemini-1.5-pro"),
        }
    
    def is_ollama_available(self) -> bool:
        """Check if Ollama is running locally."""
        try:
            response = requests.get(f"{self.ollama_host}/api/tags", timeout=2)
            return response.status_code == 200
        except:
            return False
    
    def get_model(self, agent_name: str) -> str:
        """Get the appropriate model for this agent."""
        local_model, cloud_model = self.model_map.get(agent_name, ("llama3:8b", "gemini-1.5-flash"))
        
        if self.use_local and self.is_ollama_available():
            return f"ollama/{local_model}"
        return f"gemini/{cloud_model}"
    
    def generate(self, agent_name: str, prompt: str, system_prompt: str = "") -> str:
        """Generate text using the best available model."""
        model = self.get_model(agent_name)
        
        if model.startswith("ollama/"):
            return self._ollama_generate(model.replace("ollama/", ""), prompt, system_prompt)
        else:
            return self._gemini_generate(model, prompt, system_prompt)
    
    def _ollama_generate(self, model: str, prompt: str, system: str) -> str:
        """Call local Ollama API."""
        response = requests.post(
            f"{self.ollama_host}/api/generate",
            json={
                "model": model,
                "prompt": f"{system}\n\n{prompt}" if system else prompt,
                "stream": False,
                "options": {
                    "temperature": 0.7,
                    "num_predict": 1024,
                }
            },
            timeout=60
        )
        return response.json()["response"]
    
    def _gemini_generate(self, model: str, prompt: str, system: str) -> str:
        """Call Gemini API."""
        import google.generativeai as genai
        genai.configure(api_key=self.gemini_key)
        
        model = genai.GenerativeModel(model)
        response = model.generate_content(
            f"{system}\n\n{prompt}" if system else prompt
        )
        return response.text
```

```jsx
// frontend/src/components/ModelToggle.jsx
import { useState } from 'react';
import { Cpu, Cloud, Wifi, WifiOff } from 'lucide-react';

export function ModelToggle({ onToggle, isLocal }) {
  const [ollamaStatus, setOllamaStatus] = useState('checking');
  
  // Check Ollama status on mount
  useState(() => {
    fetch('http://localhost:11434/api/tags')
      .then(() => setOllamaStatus('online'))
      .catch(() => setOllamaStatus('offline'));
  });
  
  return (
    <div className="flex items-center gap-4 p-3 bg-gray-900 rounded-lg">
      <div className="flex items-center gap-2">
        {ollamaStatus === 'online' ? (
          <Wifi className="w-4 h-4 text-green-400" />
        ) : (
          <WifiOff className="w-4 h-4 text-red-400" />
        )}
        <span className="text-sm text-gray-300">
          Ollama: {ollamaStatus}
        </span>
      </div>
      
      <div className="h-4 w-px bg-gray-700" />
      
      <button
        onClick={() => onToggle(!isLocal)}
        className={`flex items-center gap-2 px-3 py-1.5 rounded text-sm font-medium transition-colors ${
          isLocal 
            ? 'bg-green-600 text-white' 
            : 'bg-blue-600 text-white'
        }`}
      >
        {isLocal ? (
          <>
            <Cpu className="w-4 h-4" />
            Local (Ollama)
          </>
        ) : (
          <>
            <Cloud className="w-4 h-4" />
            Cloud (Gemini)
          </>
        )}
      </button>
      
      {isLocal && (
        <span className="text-xs text-green-400">
          💰 Zero API cost
        </span>
      )}
    </div>
  );
}
```

```bash
# setup-ollama.sh

echo "🦙 Setting up Ollama local models..."

# Check if Ollama is installed

ollama serve &
```
